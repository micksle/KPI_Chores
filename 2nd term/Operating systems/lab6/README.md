Datas segment   ; Сегмент даних
X dw 7          ; з вихідними
Y dw 3          ; числами
W dw 11         , і змінна
Z dw ?          ; для збереження
Datas ends      
Codes segment
Assume cs:Codes, ds : Datas
First:
MOV ax, Datas
mov ds,ax
mov ax,X ; x -> ax
add ax,Y ; x+y -> ax
mul W ; (x+y)*W -> ax
sub ax,W ; (x+y)*W-W -> ax
mov Z,ax ; ax -> Z'
mov ax,X ; x -> ax
cwd
div Y ;x/y -> ax
sub Z,ax ; Z' -x/y -> Z"
mov ah,4ch
int 21h
Codes ends
end First
